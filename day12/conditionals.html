条件分岐

if else

よくある条件分岐

if (式1){
	式1が真のとき実行
} else if (式2){
	式1が負で式2が真のとき実行
} else {
	式2が負のとき実行
}

switch

switch (式1){
	case 条件1:
		条件1に合致する時実行;
		break;
	case 条件2:
		条件2に合致する時実行;//breakを書かないと下にある処理も実行する
	case 条件3:
		条件3に合致する時実行;
		break;
	default:
		全部外れたとき;
		break;
}
