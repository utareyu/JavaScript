内部のJavaScript


header内に直接スクリプトを書くタイプ
多分もう使わない
DOMContentLoadedを付けるとbodyが完全に解析された後にスクリプトの実行が始まる
(たぶんasyncと同じ読み込まれ方をする)

InternalJavascript.html 
<script>
	document.addEventListener("DOMContentLoaded", function() {
		function createParagraph() {
			let para = document.createElement('p');
			para.textContent = 'ボタンが押されました!';
			document.body.appendChild(para);
		}

		const buttons = document.querySelectorAll('button');

		for(let i = 0; i < buttons.length ; i++) {
			buttons[i].addEventListener('click', createParagraph);
		}
	});
</script>

参考: https://developer.mozilla.org/ja/docs/Learn/JavaScript/First_steps/What_is_JavaScript#internal_javascript
